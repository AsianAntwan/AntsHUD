//Health Bonus Pulse
event HudHealthBonusPulse
{
	Animate PlayerStatusHealthValue 		FgColor 	"Health Normal" 	Linear 0.0 0.0
	Animate PlayerStatusHealthValueSpec 	FgColor 	"Health Normal" 	Linear 0.0 0.0
	Animate PlayerStatusHealthValueSpecgui 	FgColor 	"Health Normal" 	Linear 0.0 0.0
	Animate PlayerStatusHealthValueFreezePanel 	FgColor 	"Health Normal" 	Linear 0.0 0.0
	
	Animate	HealthBG			BgColor		"Health Buff"		Accel 0.0 0.0
    Animate	HealthBG			Alpha		"255"			Accel 0.0 0.3
	Animate	HealthBG			Alpha		"255"				Accel 0.3 0.6
	Animate	TargetHealthBG		BgColor		"Health Buff"		Accel 0.0 0.0
    Animate	TargetHealthBG		Alpha		"255"			Accel 0.0 0.3
	Animate	TargetHealthBG		Alpha		"255"				Accel 0.3 0.6
	Animate	SpecHealthBG		BgColor		"Health Buff"		Accel 0.0 0.0
    Animate	SpecHealthBG		Alpha		"255"			Accel 0.0 0.3
	Animate	SpecHealthBG		Alpha		"255"				Accel 0.3 0.6
	Animate	FreezeHealthBG		BgColor		"Health Buff"		Accel 0.0 0.0
    Animate	FreezeHealthBG		Alpha		"255"			Accel 0.0 0.3
	Animate	FreezeHealthBG		Alpha		"255"				Accel 0.3 0.6
	
	RunEvent HudHealthBonusPulseLoop	0.4
}

// Flash the medic charge hud when we have full charge

event HudMedicCharged
{
	Animate ChargeLabel 	FgColor 	"AntsHUDWhite" 	Linear 0.0 0.0
	Animate ChargeLabel 	FgColor 	"AntsHUDWhite" 	Linear 0.0 0.0
	
	Animate	ChargeMeter 	FgColor		"Flashing Uber Color1"		Linear 0.0 0.3
	Animate	ChargeMeter 	FgColor		"Flashing Uber Color2"		Linear 0.3 0.3

	RunEvent HudMedicChargedLoop	0.6
}

// call to loop HudHealthBonusPulse

{

}

{

// End of Casual Match Survey
event SurveyShowSequence
{
	Animate Survey ypos 480 Linear 0 0
	Animate Survey ypos 150 Linear 1 0.3
	Animate Survey ypos 130 Deaccel 1.3 0.1
	Animate Survey ypos 150 Accel 1.4 0.1
}

event SurveySubmitSequence
{
	// Show submitting
	SetVisible QuestionContainer 0 0
	SetVisible SubmittingContainer 1 0

	// Show thanks
	SetVisible SubmittingContainer 0 2
	SetVisible ThanksContainer 1 2

RunEvent SurveyHideSequence 6
}

event SurveyDelete
{
	FireCommand 0 "delete"
}

event SurveyHideSequence
{
	SetVisible CloseButton 0 0 

	Animate Survey ypos 130 Deaccel 0 0.3
	Animate Survey ypos 480 Accel 0.3 0.2
	RunEventChild Survey SurveyDelete 0.6
}
